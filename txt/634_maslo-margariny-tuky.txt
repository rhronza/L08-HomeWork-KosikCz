            var filters = $('.filter-more-result');
            var defaultFilterInputs = {"brand":[101,129,108,162,126,141,128,105,151,1486]};

            function hideCheckboxes(rows, key) {

                if (rows.legnth <= 10) {
                    return;
                }

            rows.each(function (i, e) {
                var val = +$(e).find("input").val();
                var arr = defaultFilterInputs[key];

                if ($.inArray(val, arr) === -1) {
                    $(e).hide();
                }
            });
        }

            $('.' + buttonsShowMoreClass).on('click', function (e) {
                e.stopPropagation();

                var filterCategories = $(this).closest('.' + filterCategoriesClass);

                for (var key in defaultFilterInputs) {
                    if (defaultFilterInputs.hasOwnProperty(key)) {
                        if (filterCategories.attr("id") === "filter-" + key) {
                            continue;
                        }
                        var rows = $('#filter-' + key + ' .' + formRowClass);
                        hideCheckboxes(rows, key);
                        $('#filter-' + key + ' .' + buttonsShowMoreClass).show();
                    }
                }

                filterCategories
                        .find('.' + formRowClass)
                        .show();

                $('.' + buttonsShowLessClass).hide();

                filterCategories
                        .find('.' + buttonsShowLessClass)
                        .show();

                $(this).hide();
            });


            $('.' + buttonsShowLessClass).on('click', function (e) {
                e.stopPropagation();

                var filterCategories = $(this).closest('.' + filterCategoriesClass);
                var key = filterCategories.attr("id").split("-")[1];
                var rows = filterCategories.find('.' + formRowClass);

                hideCheckboxes(rows, key);

                filterCategories

                        .show();

                $(this).hide();
            });

                e.stopPropagation();

            });
